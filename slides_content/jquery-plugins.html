jQuery Plugins

Son método que se utilizan para extender el prototipo del objeto jQuery.

Conceptos básicos

Extender las funciones de jQuery 

$.fn.colorear = function() {
    this.css("color","green";
};
$(".coloreable").colorear();

Encadenamiento 

$.fn.colorear = function() {
    this.css("color", "green");
    return this;
};
$(".coloreable").colorear().addClass(".coloreado");

Proteger el alias $ y añadir ámbito 

(function( $ ){
    $.fn.colorear = function() {
        this.css("color", "green");
        return this;
    };
}( jQuery ));

Variables privadas 

(function( $ ){
    var tono = "green";
    $.fn.colorear = function() {
        this.css( "color", tono );
        return this;
    };
}( jQuery ));

Minimizar la huella del plugin

Mal 

( function( $ ) {
    $.fn.colorearVerde = function(){ // Código para colorear verde. };
    $.fn.colorearRojo= function(){ // Código para colorear rojo. };
}( jQuery ));

Bien 

(function( $ ) {
    $.fn.colorear = function( color ) {
        if ( color === "verde") { // Código para colorear verde. }
        else if ( color === "rojo") { // Código para colorear rojo. }
    }
}( jQuery ));

Usar el método each 

$.fn.colorear = function() {
    return this.each( function() {
        // Hacer algo a cada elemento aquí.
    });
};

Aceptar opciones 


( function ( $ ){
    $.fn.colorear = function( options ) {
        // Esta es la forma más facil de tener opciones por defecto.
        var settings = $.extend({ // Estos son los de por defecto.
            color: "green",
            background: "white"
        }, options );
        // Colorear la colección usando la variable de configuración.
        return this.css({
            color: settings.color,
            background: settings.background
        });
    };
}( jQuery ));

$(".coloreable").colorear({color: "orange"});

Conceptos avanzados

Acceso público a los ajustes predeterminados Plugin 

$.fn.colorear = function( options ) {
    // Extendemos nuestras opciones por defecto con las que pasan.
    // Nota que el primer argumento a extender es un objeto vacío
    // Esto es para evitar sobre escribir nuestro objeto defaults.
    var opts = $.extend( {}, $.fn.colorear.defaults, options );
    // La implementación del plugin va aquí
};

// Plugin defaults – los añadimos como propiedad de nuestro plugin.
$.fn.colorear.defaults = {
    color: "green",
    background: "white"
};

Acceso público a los ajustes predeterminados Plugin

Mal 

var opts = $.extend( {}, $.fn.colorear.defaults);
var opts = $.extend( {}, options, $.fn.colorear.defaults );

Bien 

var opts = $.extend( {}, $.fn.colorear.defaults, options );
var opts = { icon: "tree", text: "void" '});
opts = $.extend( opts, $.fn.colorear.defaults, options );

Sobreescribir ajustes predeterminados 

// Sobre escribe el color por defecto.
$.fn.colorear.defaults.color = "blue";
                                
// Llama al plugin usando los nuevos valores por defecto.
$(".coloreable").colorear();

// Sobre escribe los valores por defecto pasando opciones al plugin.
$(".coloreable").colorear({ color: "orange" });

Acceso público a funciones secundarias 

$.fn.colorear = function( options ) {
    return this.each(function() {
        var $this = $( this );
        // ...
        var markup = $this.html();
        // Llamamos a la función de formateo.
        markup = $.fn.colorear.format( markup );
        $this.html( markup );
    });
};
// Definimos nuestra función de formateo.
$.fn.colorear.format = function( txt ) {
    return "
" + txt + "
";
};

Funciones privadas 

// Creamos la closure.
(function( $ ) {
    // Definimos el plugin.
    $.fn.colorear = function( options ) {
        debug( this );
    };
    // Función privada para depuración.
    function debug( $obj ) {
        if ( window.console && window.console.log ) {
            window.console.log( "numero de objetos: " + $obj.size() );
        }
    };
 // Fin de la closure.
})( jQuery );

Metadatos

    // Construir opciones principales antes iteración elemento.
    var opts = $.extend( {}, $.fn.colorear.defaults, options );

        var $this = $( this );


        // Construimos opciones específicas del elemento.
        // Extendemos nuestro objeto de opciones con los metadatos.
        // Tiene prioridad lo definido mediante data-* en el HTML.

        var o = $.extend( {}, opts, $this.data() );
    });
 
};
